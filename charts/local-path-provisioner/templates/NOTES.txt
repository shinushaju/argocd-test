You can create a hostpath-backed persistent volume with a persistent volume claim like this:

apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: local-path-pvc
spec:
  accessModes:
    - ReadWriteOnce
  storageClassName: {{ .Values.localPathProvisioner.storageClass.name }}
  resources:
    requests:
      storage: 2Gi
